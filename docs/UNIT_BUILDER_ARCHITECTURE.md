### Google Slides API Implementation

```javascript
// services/googleSlidesService.js

import { google } from 'googleapis';

class GoogleSlidesService {
  constructor(auth) {
    this.slides = google.slides({ version: 'v1', auth });
    this.drive = google.drive({ version: 'v3', auth });
  }

  /**
   * Create a new presentation for a lesson
   */
  async createPresentation(lesson, unit, theme = 'dark') {
    // Create the presentation
    const presentation = await this.slides.presentations.create({
      requestBody: {
        title: `${unit.title} - ${lesson.title}`
      }
    });

    const presentationId = presentation.data.presentationId;

    // Build all slides
    const requests = [];

    // 1. Title slide (already exists, just update it)
    requests.push(...this.buildTitleSlide(lesson, unit));

    // 2. Objective slide
    requests.push(...this.buildObjectiveSlide(lesson));

    // 3. Minds On slide
    if (lesson.mindsOn) {
      requests.push(...this.buildMindsOnSlide(lesson.mindsOn));
    }

    // 4. Vocabulary slides
    for (const term of lesson.vocabularyTerms || []) {
      requests.push(...this.buildVocabularySlide(term));
    }

    // 5. Activity instruction slides
    for (const activity of lesson.activities || []) {
      requests.push(...this.buildActivitySlide(activity));
    }

    // 6. Share Out slide
    if (lesson.shareOut) {
      requests.push(...this.buildShareOutSlide(lesson.shareOut));
    }

    // 7. Exit Ticket slide
    if (lesson.exitTicket) {
      requests.push(...this.buildExitTicketSlide(lesson.exitTicket));
    }

    // Apply theme
    requests.push(...this.applyTheme(theme));

    // Execute all requests
    await this.slides.presentations.batchUpdate({
      presentationId,
      requestBody: { requests }
    });

    return {
      presentationId,
      url: `https://docs.google.com/presentation/d/${presentationId}/edit`,
      embedUrl: `https://docs.google.com/presentation/d/${presentationId}/embed`
    };
  }

  /**
   * Build title slide requests
   */
  buildTitleSlide(lesson, unit) {
    return [
      {
        updatePageProperties: {
          objectId: 'title-slide',
          pageProperties: {
            pageBackgroundFill: {
              solidFill: { color: { rgbColor: { red: 0.1, green: 0.1, blue: 0.18 } } }
            }
          },
          fields: 'pageBackgroundFill'
        }
      },
      {
        insertText: {
          objectId: 'title-placeholder',
          text: lesson.title
        }
      },
      {
        insertText: {
          objectId: 'subtitle-placeholder',
          text: `${unit.title} â€¢ ${lesson.phase.toUpperCase()} Day ${lesson.dayNumber}`
        }
      }
    ];
  }

  /**
   * Build vocabulary slide
   */
  buildVocabularySlide(term) {
    const slideId = `vocab-${term.id}`;

    return [
      {
        createSlide: {
          objectId: slideId,
          slideLayoutReference: { predefinedLayout: 'TITLE_AND_TWO_COLUMNS' }
        }
      },
      {
        insertText: {
          objectId: `${slideId}-title`,
          text: `ğŸ“š ${term.term}`
        }
      },
      {
        insertText: {
          objectId: `${slideId}-left`,
          text: term.definition
        }
      },
      {
        insertText: {
          objectId: `${slideId}-right`,
          text: term.example || ''
        }
      }
    ];
  }

  /**
   * Apply a theme to the presentation
   */
  applyTheme(themeName) {
    const themes = {
      dark: {
        background: { red: 0.1, green: 0.1, blue: 0.18 },
        title: { red: 1, green: 1, blue: 1 },
        body: { red: 0.72, green: 0.72, blue: 0.82 },
        accent: { red: 0.6, green: 0.4, blue: 1 }
      },
      light: {
        background: { red: 1, green: 1, blue: 1 },
        title: { red: 0.1, green: 0.1, blue: 0.18 },
        body: { red: 0.3, green: 0.3, blue: 0.3 },
        accent: { red: 0.2, green: 0.4, blue: 0.8 }
      }
    };

    const theme = themes[themeName] || themes.dark;

    // Return requests to apply theme colors
    return [
      // Theme application requests would go here
    ];
  }

  /**
   * Export presentation as PDF
   */
  async exportAsPdf(presentationId) {
    const response = await this.drive.files.export({
      fileId: presentationId,
      mimeType: 'application/pdf'
    }, { responseType: 'arraybuffer' });

    return Buffer.from(response.data);
  }

  /**
   * Share presentation with students (view only)
   */
  async shareWithClass(presentationId, studentEmails) {
    for (const email of studentEmails) {
      await this.drive.permissions.create({
        fileId: presentationId,
        requestBody: {
          type: 'user',
          role: 'reader',
          emailAddress: email
        }
      });
    }
  }
}

export default GoogleSlidesService;
```

## Technical Stack Summary

| Layer | Technology | Purpose |
|-------|------------|---------|
| **Frontend** | React 18 + Vite | Unit Builder UI |
| **Styling** | Tailwind CSS | Rapid UI development |
| **State** | Zustand or Redux | Complex form state |
| **Backend** | Firebase | Auth, Firestore, Storage |
| **Slides** | Google Slides API | Slide generation |
| **PDF** | jsPDF + html2canvas | Printable materials |
| **Code Editor** | Monaco Editor | Code playground activities |
| **p5.js** | p5.js-react | Live code preview |
